   1import { timeout, delay } from './TimedPromise';
   2import { o }              from './mithril';
   3
   4o.spec("node", () => {
   5    let reject: any; 
   6    let resolve: any;
   7
   8    o.spec("timeout", () => {
   9        o.beforeEach((done:any) => { 
  10            reject = o.spy();
  11            resolve = o.spy();
  12            timeout(100)
  13                .then(resolve)
  14                .catch(reject);
  15            setTimeout(done, 101);
  16        });
  17        
  18        o('should fail after 100ms', (done:any) => {
  19            o(resolve.callCount).equals(0);
  20            o(reject.callCount).notEquals(0);
  21            done();
  22        });
  23    });
  24
  25    o.spec("delay", () => {

  27            reject = o.spy();
  28            resolve = o.spy();
  29            Promise.resolve()
  30                .then(delay(100))
  31                .then(resolve)
  32                .catch(reject);
  33            setTimeout(done, 101);
  34        });
  35        
  36        o('should resolve after 100ms', (done:any) => {
  37            o(resolve.callCount).notEquals(0);
  38            o(reject.callCount).equals(0);
  39            done();
  40        });
  41    });
  42});
