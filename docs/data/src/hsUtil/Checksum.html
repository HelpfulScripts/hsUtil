   6 /** 

   7  * fast implementation for short strings (20-500 chars)

   8  */

   9 export function shortCheckSum(s:string):string {

  10    let chk = 0x12345678;

  11    for (let i=0; i<s.length; i++) {

  12        chk += (s.charCodeAt(i) * (i + 1));

  13    }

  14    return (chk & 0xffffffff).toString(16); 

  15 }
