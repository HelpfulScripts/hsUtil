  14    /**

  15     * Update and zero pad nc

  16     */

  17    updateNC():string {

  18        let max = 99999999;

  19        let padding = new Array(8).join('0') + '';

  20        this.nc = (this.nc > max ? 1 : this.nc + 1);

  21        let nc = this.nc + '';

  22        return padding.substr(0, 8 - nc.length) + nc;

  23    }

  25    /**

  26     * Parse challenge digest

  27     * @param qop 

  28     */

  29    generateCNONCE(qop:string) {

  30        let cnonce:any;

  31        let nc:string;

  32

  33        if (typeof qop === 'string') {

  34            let cnonceHash = createHash('md5');

  35            cnonceHash.update(Math.random().toString(36));

  36            cnonce = cnonceHash.digest('hex').substr(0, 16);

  37            nc = this.updateNC();

  38        }

  39        return {cnonce: cnonce, nc: nc};

  40    }

  42    /**

  43     * 

  44     * @param options 

  45     * @param data 

  46     * @param response 

  47     */

  48    testAuth(options:Options, data:string, response:IncomingMessage): Options {

  49        log.debug(()=>`received www-authenticate request for ${options.host}`);

  50

  51        let challenge:any = parseDigestResponse(response.headers['www-authenticate']);

  52        let ha1 = createHash('md5');

  53        let _str = `${this.username}:${challenge.realm}:${this.password}`;

  54        ha1.update(_str);

  55        let ha2 = createHash('md5');

  56        _str = `${options.method}:${options.path}`;

  57        ha2.update(_str);

  58    

  59        let {nc, cnonce} = this.generateCNONCE(challenge.qop);

  60        let hash = createHash('md5');

  61        _str = `${ha1.digest('hex')}:${challenge.nonce}:${nc}:${cnonce}:${challenge.qop}:${ha2.digest('hex')}`;

  62        hash.update(_str);

  63    

  64        // Setup response parameters

  65        let authParams:any = {

  66            realm: challenge.realm,

  67            username: this.username,

  68            uri: options.path,

  69            qop: challenge.qop,

  70            nonce: challenge.nonce,

  71            algorithm: "MD5",

  72            response: hash.digest('hex')

  73        };

  74    

  75        authParams = omitNull(authParams);

  76    

  77        if (cnonce) {

  78            authParams.nc = nc;

  79            authParams.cnonce = cnonce;

  80        }

  81    

  82        options.headers.Authorization = compileParams(authParams);

  83        return options;

  84    }

  87function omitNull(data:any) {

  88    let newObject = {};

  89    Object.keys(data).forEach((k:string) => {

  90        if (data[k] !== null && data[k] !== undefined) { newObject[k] = data[k]; }

  91    });

  92    return newObject;

  93}

  95/**

  96 * Compose authorization header

  97 * @param params 

  98 */

  99function compileParams(params:any) {

 100    let parts = [];

 101    for (let i in params) {

 102        if (typeof params[i] !== 'function') {

 103            let param = i + '=' + (putDoubleQuotes(i) ? `"${params[i]}"` : params[i]);

 104            parts.push(param);

 105        }

 106    }

 107    return 'Digest ' + parts.join(',');

 108}

 110/**

 118function parseDigestResponse(digestHeader:string) {

 119    digestHeader = digestHeader.split('Digest ')[1];

 120    const params = {};

 121    // digestHeader.replace(/(([^"]|"[^"]*")*?)(,)/g, (...rest:string[]) => {

 124            if (part) {

 126            params[kv[0]] = kv[1].replace(/\"/g, '');

 128        return '';

 129    });
