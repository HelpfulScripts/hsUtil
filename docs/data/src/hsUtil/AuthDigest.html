  14    /**

  15     * Update and zero pad nc

  16     */

  17    updateNC():string {

  18        let max = 99999999;

  19        let padding = new Array(8).join('0') + '';

  20        this.nc = (this.nc > max ? 1 : this.nc + 1);

  21        let nc = this.nc + '';

  22        return padding.substr(0, 8 - nc.length) + nc;

  23    }

  25    /**

  26     * Parse challenge digest

  27     * @param qop 

  28     */

  29    generateCNONCE(qop:string, cnonce?:string) {

  30        if (!qop || qop === 'auth') {

  31            let cnonceHash = createHash('md5');

  32            cnonceHash.update(Math.random().toString(36));

  33            cnonce = cnonce ?? cnonceHash.digest('hex').substr(0, 16);

  34        }

  35        return {cnonce: cnonce, nc: this.updateNC()};

  36    }

  38    /**

  39     * 

  40     * @param options 

  41     * @param data 

  42     * @param response 

  43     */

  44    testAuth(options:Options, response:IncomingMessage, CNonce?:string): Options {

  45        const challenge:any = parseDigestResponse(response.headers['www-authenticate']);

  46        const ha1 = createHash('md5');

  47        const ha2 = createHash('md5');

  48        const hash = createHash('md5');

  49        ha1.update(`${this.username}:${challenge.realm}:${this.password}`);

  50        ha2.update(`${options.method}:${options.path}`);

  51        const {nc, cnonce} = this.generateCNONCE(challenge.qop, CNonce);

  52        hash.update(`${ha1.digest('hex')}:${challenge.nonce}:${nc}:${cnonce}:${challenge.qop}:${ha2.digest('hex')}`);

  53    

  54        // Setup response parameters

  55        let authParams:any = {

  56            realm: challenge.realm,

  57            username: this.username,

  58            uri: options.path,

  59            qop: challenge.qop,

  60            nonce: challenge.nonce,

  61            algorithm: "MD5",

  62            response: hash.digest('hex')

  63        };

  64    

  65        authParams = omitNull(authParams);

  66    

  67        if (cnonce) {

  68            authParams.nc = nc;

  69            authParams.cnonce = cnonce;

  70            log.info(`digest nc=${nc}`)

  71        }

  72    

  73        options.headers.Authorization = compileParams(authParams);

  74        options.headers.Accept = 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8'

  75        options.headers['Accept-Encoding'] = 'gzip, deflate';

  76        return options;

  77    }

  80function omitNull(data:any) {

  81    let newObject = {};

  82    Object.keys(data).forEach((k:string) => data[k]? newObject[k] = data[k] : '');

  83    return newObject;

  84}

  86/**

  87 * Compose authorization header

  88 * @param params 

  89 */

  90function compileParams(params:any) {

  91    const putDoubleQuotes = (entry:string) => ['qop', 'nc'].indexOf(entry)<0;

  92    let parts = [];

  93    for (let i in params) {

  94        if (typeof params[i] !== 'function') {

  95            let param = i + '=' + (putDoubleQuotes(i) ? `"${params[i]}"` : params[i]);

  96            parts.push(param);

  97        }

  98    }

  99    return 'Digest ' + parts.join(',');

 103function parseDigestResponse(digestHeader:string) {

 104    digestHeader = digestHeader.split('Digest ')[1];

 105    const params = {};

 106    digestHeader.replace(/([^,]*)/g, (...rest:string[]) => {

 107        const part = rest[1];

 108        if (part) {

 109            const kv = part.split('=').map((v:string) => v.trim());

 110            params[kv[0]] = kv[1].replace(/\"/g, '');

 111        }

 112        return '';

 113    });

 114    return params;
