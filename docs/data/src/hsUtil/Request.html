Request.ts

   1/**

   2 * Utility functions for HTTP and HTTPS GET, PUT, and POST. The module wraps a standard `XMLHttpRequest` and provides support for

   3 * - simple authentication methods, 

   4 * - decoding of received content.

   5 * - request pacing,

   6 * 

   7 * ### Authentication

   8 * Authentication is enabled by setting {@link Request.Request.setCredentials `request.setCredentials`} before any call to 

   9 * {@link Request.Request.get `request.get`} or {@link Request.Request.post `request.post`}.

  10 * The following authentication schemes are currently supported:

  11 * - Basic

  12 * - Digest

  13 * 

  14 * ### Decoding

  15 * Content received from a server or a cache can be decoded before returning the result. 

  16 * Decoding is disabled by default and can be enabled by setting a `Decoder` function:

  17 * `request.decode = Request.decoders.str2json`

  18 * 

  19 * For convenience, the following predefined decoder functions are provided as static members of `Request.decoders`:

  20 * - str2json applies JSON.parse to the content to return an object literal

  21 * - html2json applies (html2json)[https://github.com/jxck/html2json#readme] to the content

  22 * 

  23 * ### Pacing

  24 * Enable request pacing via `request.setPace(ms, max)`. When enabled, requests will be issued with a minimal 

  25 * amount of `ms` milliseconds between them, and there will never be more than `max` number of calls pending.

  26 * All additional calls will be placed in a queue until the `ms` abd `max` conditions allow calling them.

  27 * 

  28 * ### Usage:

  29 * ```

  30 * import { Request } from 'hsutil';

  31 * const request = new Request.Request();

  32 * request.cache = __dirname + '/../_data/cache';

  33 * request.decode = (options:http.Options) => options.path.indexOf('binData?id=') < 0;

  34 * request.setAuthToken(apptokens.token);

  35 * const content = await request.get('http://...');

  36 * ```

  37 */

 101export class Authenticators {

 102    private static auths:{[url:string]:Auth} = {}

 103    private static authProviders:{name:string, provider:(credentials:Credentials)=>Auth}[] = []

 104    public static addAuthProvider(name:string, provider:(credentials:Credentials)=>Auth) {

 105        Authenticators.authProviders.push({name:name, provider:provider})

 106    }

 107    public static get(authRequest:string, credentials:Credentials, url:string):Auth {

 108        if (!authRequest) { // no request for authorization: return undefined

 109            return undefined

 110        } else if (!credentials) { 

 111            throw `credentials missing; call 'setCredentials' before calling 'get' for url ${url}`;

 112        }

 113        let auth = Authenticators.auths[url];

 114        if (!auth) {

 115            if (!Authenticators.authProviders.some(entry => {

 116                if (authRequest.indexOf(entry.name)===0) return auth = entry.provider(credentials)

 117            })) throw `unimplemented authentication method ${authRequest} for '${url}'; ensure the proper authentication model is imported`;

 118        }

 119        return auth

 120    }

 121}

 232    protected getOptions(url:string|URL, method:Method):Options {

 233        url = this.getURL(url);

 234        const options = {

 235            rejectUnauthorized: false,

 236            method:     method,

 237            protocol:   url.protocol,

 238            host:       url.host,

 239            hostname:   url.hostname,

 240            port:       url.port,

 241            pathname:   url.pathname,

 242            path:       url.pathname + (url.search || ''),

 243            headers:    <any>{},

 244            url:        `${url.protocol}//${url.host}${url.pathname + (url.search || '')}`,

 245        };  

 246        if (this.authToken) { options.headers.AuthToken = this.authToken; }

 247        return options;  

 248    }

 250    /**

 251     * issues a `GET` or `POST` call and returns the `url` response. 

 252     * If `usedCache` is `true`, the cached version will be returned if available. 

 253     * Otherwise an `http` or `https` request is initiated, the result cached and returned.

 254     * If `decode` is defined it is called with the retrieved data before returning the result. 

 255     * @param url the url to retrieve

 256     * @param method 'GET' or 'POST'

 257     * @param useCached if `true`, attempt to return a cahced version. If `false`, or no cached version

 258     * exists, a remote call will be attempted.

 259     * @param postData data to submit for a `POST` call

 260     */

 261    protected async decodedRequest(options:Options, useCached:boolean, headers:any, postData?:any):Promise<Response> {

 262        const result = await this.requestOptions(options, useCached, postData); 

 263        if (this.decode && result.response.txt && options.method==='GET') { try {

 264            result.data = this.decode(<string>result.data, options);

 265        } catch(e) {

 266            this.log.error(`decoding ${result.cached?'cached ':''}${options.url}:`)

 267            this.log.error(`${e}`)

 268        }}

 269        return result;

 270    }

 272    protected async requestOptions(options:Options, useCached:boolean, postData?:any):Promise<Response> {

 273        let request: Promise<Response>;

 274        if (this.pace) {

 275            this.log.transient(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) ${options.method}-ing ${options.url}`); 

 276            request = this.pace.add(() => this.request(options, postData), `${options.method} ${options.url}`);

 277        } else {

 278            request = this.request(options, postData);

 279        }

 280        // this.log.debug(()=>`${options.method}-ing ${options.url}`); 

 281        const response = await request;

 282        const httpCode = response.response.statusCode||response.response.status;

 283        const queueInfo = this.pace? `(${this.pace.inQueue()} | ${this.pace.inProgress()}) ` : ''

 284        this.log.transient(`${queueInfo}received ${options.method} ${httpCode} ${options.url} `);

 285        return response;

 286    }

 288    protected async request(options:Options, postData?:any):Promise<Response> {

 289        let response:Response;

 290        try {

 291            response = await this.issueRequest(options, postData);        

 292            const auth = Authenticators.get(response.response.headers['www-authenticate'], this.credentials, options.url);

 293            response =  auth? await this.request(auth.testAuth(options, response.response)) : response;

 294        } catch(e) {

 295            this.log.error(`request: ${e}\n${Object.entries(options.headers).join('\n')}`)

 296            throw `error requesting ${options.url}: ${e}`;

 297        }

 298        const len = typeof response.data === 'string'? response.data.length : response.data.byteLength

 299        this.log.info(`received ${options.method} ${response.response.statusCode||response.response.status} ${(''+len).padStart(5,' ')} bytes ${options.url} `);

 300        return response

 301    }

 303    protected async issueRequest(options:Options, postData?:any):Promise<Response> {

 304        const isTextualContent = this.isTextualContent.bind(this)

 305        return new Promise((resolve:(out:Response)=>void, reject:(e:Response)=>void) => { try {

 306            function confirmRequest(e:any) {

 307                const headersText = xhr.getAllResponseHeaders();

 308                const headers = {'content-type':''};

 309                headersText.split('\r\n').map(h => {

 310                    const kv = h.split(':').map(p => p.trim());

 311                    if (kv[0].length) { headers[kv[0]] = kv[1]; }

 312                })

 313                // const contentType = this.responseType;

 314                // const txt = isTextualContent(contentType);

 315                const txt = isTextualContent(headers['content-type'], headers['content-length'], options.url);

 316

 317                // const data = this.responseType==='arraybuffer'? (txt? await this.response.text() : await this.response.arrayBuffer()) : this.response;

 318                const data = this.response;

 319                const response:Response = {

 320                    data:               data,

 321                    response: {

 322                        txt:            txt,

 323                        headers:        headers,

 324                        method:         options.method,

 325                        status:         this.status, 

 326                        statusCode:     this.status, 

 327                        statusMessage:  `${this.status} ${this.statusText}`,

 328                        url:            this.responseURL,

 329                    }

 330                };

 331                resolve(response);

 332            }

 333            const xhr = new XMLHttpRequest();

 334            const txt = Request.isTextualRequest(options.pathname);

 335            // this.log.debug(()=>`requesting ${options.method} ${this.log.inspect(options, {depth:4})}`);

 336            xhr.open(options.method, options.url, true);

 337            Object.keys(options.headers).forEach(h => xhr.setRequestHeader(h, options.headers[h]));

 338            xhr.responseType = txt? 'text' : 'arraybuffer';

 339            xhr.onload = confirmRequest;

 340            xhr.send(postData?JSON.stringify(postData):undefined);

 341        } catch(e) {

 342            reject(e);

 343        }})

 344    }

 346    protected isTextualContent(contentType:string, contentLength:number, url:string):boolean {

 347        if (contentType===undefined) {

 348            this.log.warn(`undefined contentType; assuming binary for len=${contentLength} for '${url}'`)

 349            return false;

 350        }

 351        const subTypes = contentType.split('/');

 352        const match = Request.contentTypes.filter(entry => entry.subTypes.every((st, i) => subTypes[i] === st));

 353        if (match.length>0) {

 354            return match[0].isText;

 355        } else {

 356            this.log.warn(`no match found for '${contentType}'; caching as binary`);  

 357            return false;

 358        }

 359    }

 362        return ['json', 'txt', 'html'].some(ext => pathName.indexOf(ext) >= 0);
