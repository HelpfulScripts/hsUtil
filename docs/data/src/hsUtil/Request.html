Request.ts

   1/**

   2 * Utility functions for HTTP and HTTPS GET, PUT, and POST. The module wraps a standard `XMLHttpRequest` and provides support for

   3 * - simple authentication methods, 

   4 * - decoding of received content.

   5 * - request pacing

   6 * 

   7 * ### Authentication

   8 * Authentication is enabled by setting {@link Request.Request.setCredentials `request.setCredentials`} before any call to 

   9 * {@link Request.Request.get `request.get`} or {@link Request.Request.post `request.post`}.

  10 * The following authentication schemes are currently supported:

  11 * - Basic

  12 * - Digest

  13 * 

  14 * ### Decoding

  15 * Content received from a server or a cache can be decoded before returning the result. 

  16 * Decoding is disabled by default and can be enabled by setting a `Decoder` function:

  17 * `request.decode = Request.decoders.str2json`

  18 * 

  19 * For convenience, the following predefined decoder functions are provided as static members of `Request.decoders`:

  20 * - str2json applies JSON.parse to the content to return an object literal

  21 * - html2json applies (html2json)[https://github.com/jxck/html2json#readme] to the content

  22 * 

  23 * ### Pacing

  24 * Enable request pacing via `request.setPace(ms, max)`. When enabled, requests will be issued with a minimal 

  25 * amount of `ms` milliseconds between them, and there will never be more than `max` number of calls pending.

  26 * All additional calls will be placed in a queue until the `ms` abd `max` conditions allow calling them.

  27 * 

  28 * ### Usage:

  29 * ```

  30 * import { Request } from 'hsutil';

  31 * const request = new Request.Request();

  32 * request.cache = __dirname + '/../_data/cache';

  33 * request.decode = (options:http.Options) => options.path.indexOf('binData?id=') < 0;

  34 * request.setAuthToken(apptokens.token);

  35 * const content = await request.get('http://...');

  36 * ```

  37 */

 211    protected getOptions(url:string|URL, method:Method):Options {

 212        url = this.getURL(url);

 213        const prefix = /*(method==='POST')? `/cgi/save.js/` :*/ '';

 214        const options = {

 215            rejectUnauthorized: false,

 216            method:     method,

 217            protocol:   url.protocol,

 218            host:       url.host,

 219            hostname:   url.hostname,

 220            port:       url.port,

 221            pathname:   prefix + url.pathname,

 222            path:       prefix + url.pathname + (url.search || ''),

 223            headers:    {},

 224            url:        `${url.protocol}//${url.host}${url.port?':'+url.port : ''}${prefix + url.pathname + (url.search || '')}`,

 225        };  

 226        if (this.authToken) { options.headers.AuthToken = this.authToken; }

 227        return options;  

 228    }

 230    /**

 231     * issues a `GET` or `POST` call and returns the `url` response. 

 232     * If `usedCache` is `true`, the cached version will be returned if available. 

 233     * Otherwise an `http` or `https` request is initiated, the result cached and returned.

 234     * If `decode` is defined it is called with the retrieved data before returning the result. 

 235     * @param url the url to retrieve

 236     * @param method 'GET' or 'POST'

 237     * @param useCached if `true`, attempt to return a cahced version. If `false`, or no cached version

 238     * exists, a remote call will be attempted.

 239     * @param postData data to submit for a `POST` call

 240     */

 241    protected async decodedRequest(options:Options, useCached:boolean, headers:any, postData?:any):Promise {

 242        const result = await this.requestOptions(options, useCached, postData); 

 243        if (this.decode && result.response.txt && options.method==='GET') {

 244            result.data = this.decode(result.data, options);

 245        }

 246        return result;

 247    }

 264    protected async requestOptions(options:Options, useCached:boolean, postData?:any):Promise {

 265        const fname = this.cache? this.cacheName(options) : undefined;

 266        if (fname && useCached && options.method === 'GET') { 

 267            const result = await this.readCached(fname); 

 268            if (result !== undefined) { return result; }

 269        }

 270

 271        let request: Promise;

 272        if (this.pace) {

 273            this.log.info(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) ${options.method}-ing ${options.url}`); 

 274            request = this.pace.add(() => this.request(options, postData), `${options.method} ${options.url}`);

 275        } else {

 276            request = this.request(options, postData);

 277        }

 278        this.log.debug(()=>`${options.method}-ing ${options.url}`); 

 279        const response = await request;

 280        if (this.pace) { this.log.transient(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) received ${options.method} ${options.url} `); }

 281        this.log.debug(()=>`received ${options.method} ${response.response.statusMessage} ${options.method} ${options.url}`); 

 282

 283        const code = response.response.statusCode||response.response.status;

 284        if(code >= 200 && code < 300) {

 285            if (fname && options.method === 'GET') {

 286                await this.writeCached(fname, response);

 287            }

 288        }

 289        return response;

 290    }

 292    protected async request(options:Options, postData?:any):Promise {

 293        try {

 294            const response = await this.issueRequest(options, postData);        

 295            let authenticate = response.response.headers['www-authenticate'];

 296            if (!authenticate) { 

 297                return response;

 298            } else if (this.credentials) { 

 299                    authenticate = authenticate.trim();

 300                    let auth: Auth;

 301                    if (authenticate.indexOf('Digest') === 0) {

 302                        auth = new AuthDigest(this.credentials.user, this.credentials.password);

 303                    } else if (authenticate.indexOf('Basic') === 0) {

 304                        auth = new AuthBasic(this.credentials.user, this.credentials.password);

 305                    } else {

 306                        throw `unimplemented authentication request ${authenticate} for '${options.url}'`;

 307                    }

 308                    auth.testAuth(options, response.data, response.response);

 309                    return this.request(options);

 310            } else {

 311                throw `authentication missing; call 'setCredentials' before calling 'get'`;

 312            }

 313        } catch(e) {

 314            throw `error requesting ${options.url}: ${e}`;

 315        }

 316    }

 318    protected async issueRequest(options:Options, postData?:any):Promise {

 319        const request = this;

 320        return new Promise((resolve:(out:Response)=>void, reject:(e:Response)=>void) => { try {

 321            function confirmRequest(e:any) {

 322                const headersText = xhr.getAllResponseHeaders();

 323                const headers = {'content-type':''};

 324                headersText.split('\r\n').map(h => {

 325                    const kv = h.split(':').map(p => p.trim());

 326                    if (kv[0].length) { headers[kv[0]] = kv[1]; }

 327                })

 328                const contentType = this.responseType;

 329                const txt = request.isTextualContent(contentType);

 330

 331                // const data = this.responseType==='arraybuffer'? (txt? await this.response.text() : await this.response.arrayBuffer()) : this.response;

 332                const data = this.response;

 333                const response:Response = {

 334                    data:               data,

 335                    response: {

 336                        txt:            txt,

 337                        headers:        headers,

 338                        method:         options.method,

 339                        status:         this.status, 

 340                        statusCode:     this.status, 

 341                        statusMessage:  `${this.status} ${this.statusText}`,

 342                        url:            this.responseURL,

 343                    }

 344                };

 345                resolve(response);

 346            }

 347            const xhr = new XMLHttpRequest();

 348            const txt = request.isTextualRequest(options.pathname);

 349            this.log.debug(()=>`requesting ${options.method} ${this.log.inspect(options, {depth:4})}`);

 350            xhr.open(options.method, options.url, true);

 351            Object.keys(options.headers).forEach(h => xhr.setRequestHeader(h, options.headers[h]));

 352            xhr.responseType = txt? 'text' : 'arraybuffer';

 353            xhr.onload = confirmRequest;

 354            xhr.send(postData?JSON.stringify(postData):undefined);

 355        } catch(e) {

 356            reject(e);

 357        }})

 358    }

 360    protected isTextualContent(contentType:string):boolean {

 361        let txt = false;

 362        if (contentType===undefined) { contentType = 'text/html'; }

 363        else { contentType = contentType.split(';')[0]; }

 364        const subTypes = contentType.split('/');

 365        switch (subTypes[0]) {

 366            case '':       

 367            case 'text':        txt = true; break;

 368            case 'image':       

 369            case 'audio':

 370            case 'font':        break;

 371            case 'application': switch(subTypes[1]) {

 372                case 'json':    txt = true; break;

 373                case 'pdf':     break;

 374                case 'vnd.openxmlformats-officedocument.presentationml.presentation': break;

 375                case 'vnd.openxmlformats-officedocument.spreadsheetml.sheet': break;

 376                case 'vnd.ms-powerpoint': break;

 377                case 'vnd.ms-excel': break;

 378                case 'octet-stream': break;

 379                case 'vnd.ms-excel.sheet.macroenabled.12': break;

 380                default: this.log.info(`caching ${contentType} as binary`);

 381            }

 382            break;

 383            default: this.log.warn(`caching '${contentType}' as binary`);    

 384        }

 385        return txt;

 386    }

 389        return ['json', 'txt', 'html'].some(ext => pathName.indexOf(ext) >= 0);

 390    }
