   7/**

   9 * @param {number} ms the milliseconds to wait before rejecting

  10 * @return {Promise} a Promise that rejects after `ms`

  12export function timeout(ms:number):Promise { 

  13    return new Promise((resolve, reject) => { setTimeout(reject, ms); }); 

  14}

  16/**

  17 * @description delays a promise, passing the parameter 

  18 * received from the calling promise down to the resolving promise.

  19 * 

  20 * **Usage 1:** a delay within the `then` chain:

  21 * ```

  22 * 

  23 *    .then(...)

  24 *    .then(delay(10))

  25 *    .then(...)

  26 *    .catch(...)

  27 * ```

  28 * **Usage 2:** an initial delay

  29 * ```

  30 * delay(10)()

  31 *    .then(...)

  32 *    .catch(...)

  33 * ```

  34 * @param number ms the milliseconds to wait before resolving

  35 * @return a `Promise` that resolves after `ms` 

  36 */

  37export function delay(ms:number)   { 

  38    return (args?:T):Promise => {

  39        return new Promise((resolve:(args:T)=>void) => { 

  40            setTimeout(() => { resolve(args); }, ms); 

  41        }); 

  42    };

  43}

  45/**

  46 * @description ensures that function calls in a sequence are not executed faster than a preset minimum delay.

  47 * 

  48 * **Usage:** 

  49 * ```

  50 * const q = new PacingQueue(100); // 100ms between calls

  51 * ...

  52 * q.add((ms) => `I have been called after ${ms}ms`;})

  53 *    .then((result) => console.log(result));     // prints: I have been called after ***ms`

  54 * ```

  55 */

  56export class Pace {

  57    private maxConcurrent   = -1;   // max number of concurrent requests being called

  58    private pace:number;      // the pace of calls in ms

  59    private waitUntil       = 0;  // the earliest time for the next call

  60    private waitCount       = 0;  // number of calls currently in queue waiting

  61    private beingCalled     = 0;  // registered functiuons that have been called , but have not resolved yet.

  62

  63    /**

  64     * @param delay the minimum number of milliseconds between executions of 

  65     * two registered functions; defaults to 100;

  66     * @param maxConcurrent the maximum number of concurrent execution calls; defaults to -1 (no limit)

  67     */

  68    constructor(pace=100, maxConcurrent=-1) {

  69        this.pace = pace+5; // add 5ms margin. delay() may trigger a millisecond or two early

  70        this.maxConcurrent = maxConcurrent;

  71    }

  72

  73    getWaitCount()    { return this.waitCount; }

  74    getCallingCount() { return this.beingCalled; }

  75

  76    /**

  77     * adds the function to the queue. After an appropriate time has passed, 

  78     * the function will be called with the delay in ms as parameter.

  79     * @param fn 

  80     * @return a promise that resolved to the result of the function

  81     */

  82    add(fn: (msSinceAdding:number) => any):Promise {

  83// const id = Math.floor(Math.random()*10000);

  84        const addTime = Date.now();

  85        if (this.waitUntil < addTime) { this.waitUntil = addTime; }

  86        const diff = this.waitUntil - addTime;

  87        this.waitUntil += this.pace + 5;

  88        this.waitCount++;

  89// console.log(`created #${id}: wait:${this.waitCount}, called:${this.beingCalled}`);

  90        return delay(diff)()

  91        .then(async () => {

  92// console.log(`delayed #${id}: wait:${this.waitCount}, called:${this.beingCalled}`);

  93            await new Promise(resolve => {

  94                const waitLoop = () => {

  95                    if (this.maxConcurrent < 0 || this.beingCalled < this.maxConcurrent) {

  96                        resolve();

  97                    } else {

  98                        setTimeout(waitLoop, 10);

  99                    }

 100                };

 101                waitLoop();

 102            });

 103// console.log(`calling #${id}: wait:${this.waitCount}, called:${this.beingCalled}`);

 104            this.waitCount--;

 105            this.beingCalled++;

 106            return fn(Date.now()-addTime);

 107            })

 108        .then((ret:any) => {

 109            this.beingCalled--;

 110            return ret;

 111        });

 112    }

 113}

 116/**

 117 * Sequentially calls the provided `tasks` in a `.then()` chain of promises, 

 118 * guaranteeing the order of execution per index order in the array. 

 119 * Each task can return a result, or a promise for a result.

 120 * @param tasks an array of task calls to execute in sequence. Each call will pass the array of results for tasks so far executed.

 121 * @param initialResult optional initial array to collect the task results.

 122 */

 123export function promiseChain(tasks:((results:T[])=>T|Promise)[], initialResult:T[]=[]): Promise {

 124    return tasks.reduce((chain:Promise, task:(result:T[])=>T|Promise): Promise =>

 125        // execute a task by chaining it to the previous ones via `.then()`

 126        chain.then((_results:T[]) => Promise.resolve(task(_results)).then((r:T) => {

 127            // add the task's result to the `results` array

 128            _results.push(r);

 129            return _results;

 130        })), 

 131        Promise.resolve(initialResult)

 132    );

 133}
