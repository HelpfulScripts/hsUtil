   7/**

   8 * timeout promise for use in `Promise.race()`.

   9 * @param {number} ms the milliseconds to wait before rejecting

  10 * @return {Promise} a Promise that rejects after `ms` 

  11 */

  12export function timeout(ms:number):Promise { 

  13    return new Promise((resolve, reject) => { setTimeout(reject, ms); }); 

  14}

  16/**

  17 * promise-based delay function. Delays execution in a promise chain, passing the parameter 

  18 * received from the calling promise down to the resolving promise.

  19 * ```

  20 * delay(10)(param)

  21 *    .then(param => {...})

  22 *    .catch(...)

  23 * ```

  24 * or: insert delay within the `then` chain:

  25 * ```

  26 * 

  27 *    .then(a => { ... return b; })

  28 *    .then(delay(10))

  29 *    .then(b => { ... })

  30 *    .catch(...)

  31 * ```

  32 * or as an async/await pattern:

  33 * ```

  34 * try {

  35 *    await 

  36 *    ...

  37 *    await delay(10)

  38 *    ...

  39 * } catch(e) {

  40 *    ...

  41 * }

  42 * ```

  43 * @param number ms the milliseconds to wait before resolving

  44 * @return a `Promise` that resolves after `ms` 

  45 */

  46export function delay(ms:number)   { 

  47    return (args?:T):Promise => {

  48        return new Promise((resolve:(args:T)=>void) => { 

  49            setTimeout(() => { resolve(args); }, ms); 

  50        }); 

  51    };

  52}

  54/**

  55 * paces a series of similar function calls to

  56 * - not occur faster than a preset rate

  57 * - not issue more unresolved calls then a preset limit

  58 * 

  59 * **Usage:** 

  60 * ```

  61 * const q = new PacingQueue(100, 10); // 100ms between calls, at most 10 unresolved

  62 * ...

  63 * const result = await q.add((ms) => `I have been called after ${ms}ms`;})

  64 * console.log(result));     // prints: I have been called after 105ms`

  65 * ```

  66 */

  67export class Pace {

  68    /** max number of concurrent requests being called */

  69    private maxConcurrent   = -1;

  70    /** the pace of calls in ms */

  71    private pace            = 0;      

  72    /** the earliest time for the next call */

  73    private waitUntil       = 0;  

  74    /** number of calls currently waiting in queue */

  75    private waiting         = 0;  

  76    /** functions that have been called, but have not resolved yet. */

  77    private started         = 0;  

  78

  79    /**

  80     * @param pace the minimum number of milliseconds between executions of 

  81     * two registered functions; defaults to 100;

  82     * @param maxConcurrent the maximum number of concurrent execution calls; defaults to -1 (no limit)

  83     */

  84    constructor(pace=100, maxConcurrent=-1) {

  85        this.pace = pace+5; // add 5ms margin. delay() may trigger a millisecond or two early

  86        this.maxConcurrent = maxConcurrent;

  87    }

  88

  89    /** the pace of calls in ms */

  90    setPace(ms:number):void { this.pace = ms; }

  91    /** max number of concurrent requests being called */

  92    setMaxConcurrent(maxConcurrent:number):void { this.maxConcurrent = maxConcurrent; }

  93    /** number of functions currently waiting in queue */

  94    inQueue():number    { return this.waiting; }

  95    /** functions that have been called, but have not resolved yet. */

  96    inProgress():number { return this.started; }

  97

  98    /**

  99     * adds the function to the queue. After an appropriate time has passed, 

 100     * the function will be called with the delay in ms as parameter.

 101     * @param fn 

 102     * @return a promise that resolved to the result of the function

 103     */

 104    async add(fn: (msSinceAdding:number) => any):Promise {

 105        const addTime = Date.now();

 106        if (this.waitUntil < addTime) { this.waitUntil = addTime; }

 107        const diff = Math.max(0, this.waitUntil - addTime);

 108        this.waitUntil += this.pace + 5;

 109        this.waiting++;

 110        await delay(diff)();

 111        await new Promise(resolve => {

 112            const waitLoop = () => (this.maxConcurrent < 0 || this.started < this.maxConcurrent)?

 113                    resolve() : setTimeout(waitLoop, 10);

 114            waitLoop();

 115        });

 116        this.waiting--;

 117        this.started++;

 118        const ret = await fn(Date.now()-addTime);

 119        this.started--;

 120        return ret;

 121    }

 122}
