   1/**
   2 * Logging convenience functions.
   3 * ## Usage
   4 * 
   5 * ### Using the global log object 
   6 * Settings in `log` are shared across modules
   7 * 


   8 * import { log } from 'hsnode'; 

   9 * log.info('by the way:'); // -> 20160817 09:59:08.032 info by the way:

  10 * log.error('oh dear!');   // -> 20160817 09:59:08.045 error *** oh dear!

  11 * 

  12 * 
  13 * ### Using a local log object 
  14 * Settings in `log` remain local to the module 
  15 * 


  16 * import { log as gLog } from 'hsnode'; const log = gLog('myModule')

  17 * log.info('by the way:'); // -> 20160817 09:59:08.032 myModule info by the way:

  18 * log.error('oh dear!');   // -> 20160817 09:59:08.045 myModule error *** oh dear!

  19 * 

  20 * 
  21 * ### Using the `format` template:
  22 * 


  23 * log.format('%MMM %DD %hh%mm%ss');

  24 * log.info('by the way:');  // -> Aug 17 095908 info by the way:

  25 * log.error('oh dear!');    // -> Aug 17 095908 error *** oh dear!

  26 * 

  27 * 
  28 * ### Change the module prefix:
  29 * 


  30 * log.prefix('Main');

  31 * log.format('%hh%mm%ss');

  32 * log.info('by the way:');  // -> 09:59:08.032 Main info by the way:

  33 * log.error('oh dear!');    // -> 09:59:08.045 Main error *** oh dear!

  34 * 

  35 * 
  36 * ### Setting the Log level locally for the module
  37 * ```
  38 * log.level(log.WARN);
  39 * log.info('this will not be reported');
  40 * ``` 
  41 * 
  42 * ### Setting the Log level globally
  43 * Will be used if no local level is set
  44 * ```
  45 * log.level(log.WARN, true);               // set global log level
  46 * log.info('this will be reported');       // because local level defaults to INFORM
  47 * log.level(null);                         // unset the local level
  48 * log.info('this will not be reported');   // because no local level is defined and global level is set to WARN
  49 * ``` 
  50 * 
  51 * ### log.inspect utility
  52 * returns a string representing a deep inspection of `myObj`.
  53 * ```
  54 * log.info(log.inspect(myObj, null));      // prints the structure to inifinite depth
  55 * ``` 
  56 *

  58 * -   {@link log.DEBUG log.DEBUG}
  59 * -   {@link log.INFO  log.INFO}
  60 * -   {@link log.WARN  log.WARN}
  61 * -   {@link log.ERROR log.ERROR}
  62 * 
  63 * ## Reporting methods
  64 * -   {@link log.LogType.debug log.debug()}
  65 * -   {@link log.LogType.info  log.info()}

 271    function level(newLevelSym?:symbol, setGlobalLevel=false):symbol {
 272        let newLevel = gLevels[newLevelSym] || gGlobalLevel;    // new level is newLevelSym unless undefined
 273        let oldLevel = context.level || gGlobalLevel;           // old level is level unless undefined
 274        if (newLevelSym === undefined) {                        // do nothing, return current level
 275            newLevel = oldLevel;
 276        } else if (newLevelSym === null) {
 277            context.level = undefined;                          // deactivate local level
 278        } else if (gLevels[newLevelSym]) { 
 279            if (setGlobalLevel) { gGlobalLevel = newLevel; }    // set new global level
 280                           else { context.level = newLevel; }     // set new local level
 281            const msg = `new ${setGlobalLevel? 'global' : context.prefix} log level ${newLevel.desc.toUpperCase()} (was ${oldLevel.desc.toUpperCase()})`;
 282            out((newLevel.sym === oldLevel.sym)?DEBUG : INFO, msg);
 283        } else { 
 284            out(ERROR, `unkown level ${newLevelSym.toString()}; log level remains ${oldLevel.sym.toString()}`);
 285        }
 286        return newLevel.sym;
 287    }

 305    function out(lvl:symbol, msg:any, log2File=true): Promise {
 306        const colors = { [ERROR]: color.red+color.bold, [WARN]: color.yellow+color.bold, [DEBUG]: color.blue, [INFO]: color.green };
 307        let desc = gLevels[lvl];
 308        const filterLevel = context.level || gGlobalLevel;
 309        if (desc.importance >= filterLevel.importance) {
 310            const dateStr = date(gDateFormat);
 311            let line = (typeof msg === 'string')? msg : inspect(msg, 0);
 312            const logLine = (dateStr + ' ' + context.prefix + desc.desc + ' ' + line);
 313            const colorLine = `${colors[lvl]||''} ${dateStr} ${context.prefix} ${desc.desc} ${color.clear} ${line}`;
 314            console.log(gColors? colorLine : logLine);
 315            if (msg && msg.stack) { console.log(msg.stack); }
 316            if (gLogFile && log2File) {
 317                return context.logToFile(date(gLogFile), logLine);
 318            }
 319        }
 320        return Promise.resolve(undefined);
 321    }

 323    function logFile(file?:string):Promise {
 324        if (file === null) {                    // disable logging in file
 325            gLogFile = undefined; 
 326            return info("disabling logfile");
 327        } else if (file === undefined) {        // leave gLogFile unchanged, return promise for logfile name
 328            return Promise.resolve(date(gLogFile));
 329        } else if (file.indexOf('/')>=0) { 
 330            return context.pathExists(file)
 331                .then((exists:boolean) => { 
 332                    if (!exists) {
 333                        gLogFile = undefined;
 334                        return warn(`path '${file}' doesn't exists; logfile disabled`);
 335                    }
 336                    gLogFile = file;
 337                    return info("now logging to file " + date(file));
 338                })
 339                .catch(() => { 
 340                    gLogFile = undefined; 
 341                    return error(`checking path ${file}; logfile disabled`);
 342                });
 343        } else if (file === '') {
 344            file = 'log-%YYYY-%MM-%DD.txt';
 345        } else {
 346        }
 347        gLogFile=file;
 348        return info(gLogFile? `now logging to file ${date(gLogFile)}` : 'logfile disbaled');
 349    }

 358    function inspect(msg:any, depth=1, indent=''):string {
 359        if (depth===null) { depth = 999; }
 360        if (msg === null) { return 'null'; }
 361        if (typeof msg === 'function') { return 'function'; }
 362        if (typeof msg === 'string') { return `'${msg}'`; }
 363        if (typeof msg === 'object') {
 364            if (depth<0) { return (msg.length===undefined)? '{...}' : '[...]'; }
 365            if (msg.length !== undefined) { return `[${msg.map((e:any)=>(e===undefined)?'':inspect(e, depth-1, indent)).join(', ')}]`; }
 366            return '{\n' + Object.keys(msg).map(k => `   ${indent}${k}: ${
 367                inspect(msg[k], depth-1, indent+'   ')
 368            }`).join(',\n') + `\n${indent}}`;
 369        } 
 370        return msg.toString();
 371    }
