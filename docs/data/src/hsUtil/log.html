log.ts

   1/**

   2 * Logging convenience functions.

   3 * ## Usage

   4 * 

   5 * ### Using the global `Log` instance 

   6 * Settings in `Log.log` are shared across modules

   7 * ```

   8 * import { Log } from 'hsutil'; const log = Log.log;

   9 * log.level(Log.INFO);     // sets the reporting level on the global instance 

  10 * log.info('by the way:'); // -> 20160817 09:59:08.032 info by the way:

  11 * log.error('oh dear!');   // -> 20160817 09:59:08.045 error *** oh dear!

  12 * ```

  13 * 

  14 * ### Using a local `Log` instance 

  15 * Create a new `Log` instance to keep changes local to the current module:

  16 * ```

  17 * import { Log } from 'hsutil'; const log = new Log('myModule')

  18 * log.info('by the way:'); // -> 20160817 09:59:08.032 myModule info by the way:

  19 * log.error('oh dear!');   // -> 20160817 09:59:08.045 myModule error *** oh dear!

  20 * ```

  21 * 

  22 * ### Using the `format` template:

  23 * ```

  24 * log.format('%MMM %DD %hh%mm%ss');

  25 * log.info('by the way:');  // -> Aug 17 095908 info by the way:

  26 * log.error('oh dear!');    // -> Aug 17 095908 error *** oh dear!

  27 * ```

  28 * 

  29 * ### Change the module prefix:

  30 * ```

  31 * log.prefix('Main');

  32 * log.format('%hh%mm%ss');

  33 * log.info('by the way:');  // -> 09:59:08.032 Main info by the way:

  34 * log.error('oh dear!');    // -> 09:59:08.045 Main error *** oh dear!

  35 * ```

  36 * 

  37 * ### Setting the Log level locally for the module

  38 * ```

  39 * log.level(log.WARN);

  40 * log.info('this will not be reported');

  41 * ``` 

  42 * 

  43 * ### Setting the Log level globally

  44 * Will be used if no local level is set

  45 * ```

  46 * log.level(log.WARN, true);               // set global log level

  47 * log.info('this will be reported');       // because local level defaults to INFORM

  48 * log.level(null);                         // unset the local level

  49 * log.info('this will not be reported');   // because no local level is defined and global level is set to WARN

  50 * ``` 

  51 * 

  52 * ### log.inspect utility

  53 * returns a string representing a deep inspection of `myObj`.

  54 * ```

  55 * log.info(log.inspect(myObj, null));      // prints the structure to inifinite depth

  56 * ``` 

  57 * 

  58 * ## Reporting Levels:

  59 * -   {@link log.Log.DEBUG Log.DEBUG}

  60 * -   {@link log.Log.INFO  Log.INFO}

  61 * -   {@link log.Log.WARN  Log.WARN}

  62 * -   {@link log.Log.ERROR Log.ERROR}

  63 * 

  64 * ## Reporting methods

  65 * -   {@link log.Log.debug log.debug()}

  66 * -   {@link log.Log.info  log.info()}

  67 * -   {@link log.Log.warn  log.warn()}

  68 * -   {@link log.Log.error log.error()}

  69 * 

  70 * ## Configurations:

  71 * -   {@link log.Log.level   log.level()}

  72 * -   {@link log.Log.format  log.format()}

  73 * -   {@link log.Log.prefix  log.prefix()}

  74 * 

  75 * ## Inspection support:

  76 * -   {@link log.Log.inspect log.inspect()}

  77 */

 131    /**

 132     * sets the reporting level according to `newLevel`. 

 133     * Valid values are {@link log.DEBUG log.DEBUG}, {@link log.INFO log.INFO}, 

 134     * {@link log.WARN log.WARN}, {@link log.ERROR log.ERROR}, or `null`.

 135     * Subsequent reporting calls will be filtered such that only calls with an importance 

 136     * at least the same as `newLevel` will be written to the log.

 137     * 

 138     * By default, `newLevel` sets the reporting level for the module only, and takes precedence over

 139     * any global level setting. Providing `null` as level value passes precedence back to the global setting.

 140     * This allows for simple per-module settings, for example in debugging scenarios.

 141     * @param newLevelSym the new reporting level to set. 

 142     * If omitted, the method returns the currently set reporting level. 

 143     * If set to `null`, the module's local reporting level defaults to the global reporting level; `setGlobalLevel` is ignored.

 144     * @param setGlobalLevel defaults to `false`. If `true`, sets the global reporting level for all modules. 

 145     * @return the new reporting level (DEBUG, INFO, WARN, ERROR)

 146     */

 147    public level(newLevelSym?:string, setGlobalLevel?:boolean):string {

 148        let newLevel = Log.levels[newLevelSym] || Log.globalLevel;    // new level is newLevelSym unless undefined

 149        let oldLevel = this.reportLevel || Log.globalLevel;        // old level is level unless undefined

 150        if (newLevelSym === undefined) {                        // do nothing, return current level

 151            newLevel = oldLevel;

 152        } else if (newLevelSym === null) {

 153            this.reportLevel = undefined;                       // deactivate local level

 154        } else if (Log.levels[newLevelSym]) { 

 155            if (setGlobalLevel) { Log.globalLevel = newLevel; }    // set new global level

 156                           else { this.reportLevel = newLevel; }// set new local level

 157            const msg = `new ${setGlobalLevel? 'global' : this.reportPrefix} log level ${newLevel.desc.toUpperCase()} (was ${oldLevel.desc.toUpperCase()})`;

 158            this.out((newLevel.sym === oldLevel.sym)?Log.DEBUG : Log.INFO, msg);

 159        } else { 

 160            this.out(Log.ERROR, `unkown level ${newLevelSym}; log level remains ${oldLevel.sym}`);

 161        }

 162        return newLevel.sym;

 163    }

 195    /**

 196     * reports an error message to the log. 

 197     * The message will always be reported to the log.

 198     * @param msg the message to report. If msg is an object literal, a deep inspection will be printed.

 199     * @param log optional flag to enable/suppress logging to file. Defaults to `true`

 200     * @return promise to return the file written to, or undefined

 202    public async error(msg:any, log2File=true):Promise { 

 203        if (msg.message) { // special treatment for Errors

 204            await this.out(Log.ERROR, msg.message);

 205            await this.out(Log.ERROR, msg.stack);

 206            return msg.message;

 207        } else {

 208            return await this.out(Log.ERROR, msg); 

 209        } 

 210    }

 212    /**

 213     * reports an error message to the log. 

 214     * The message will be reported to the log if `lvl` meets or exceeds the current reporting level.

 215     * @param lvl the reporting level of `msg`

 216     * @param msg the message to report. If msg is an object literal, a deep inspection will be printed.

 217     * @param log optional flag to enable/suppress logging to file. Defaults to `true`

 218     * @return promise to return the message written

 219     */

 220    protected async out(lvl:string, msg:any): Promise {

 221        let desc:LevelDesc = Log.levels[lvl];

 222        const filterLevel = this.reportLevel || Log.globalLevel;

 223        if (desc.importance >= filterLevel.importance) {

 224            const dateStr = date(Log.dateFormat);

 225            let line = (typeof msg === 'string')? msg : this.inspect(msg, 0);

 226            const logLine = this.makeMessage(line, lvl, dateStr, desc.desc);

 227            console.log(logLine);

 228            if (msg && msg.stack) { console.log(msg.stack); }

 229            return Promise.resolve(line);

 230        }

 231        return undefined;

 232    }

 235     * Creates the core format of the reported message. This method is 

 236     * called within `out` to allow for format extensions of the message printed,

 239    protected makeMessage(line:string, lvl:string, dateStr:string, desc:string):string {

 240        return `${dateStr} ${this.reportPrefix} ${desc} ${line}`;

 245     * the function returns the currently set format string. The preset is '%YYYY%MM%DD %hh:%mm:%ss.%jjj'

 282    /**

 283     * Usage: `log.info(log.inspect(struct, 1))`

 284     * The call returns a raw formatted text string, or a HTML formatted string if `colors` is defined.

 285     * @param struct the object literal to inspect

 286     * @param depth depth of recursion, defaults to 3. Use `null` for infinite depth

 287     * @param indent the indent string to use, will accumulate for each level of indentation, defaults to four spaces.

 288     * @param colors an array of `css` color values: if defined, the output will be `HTML` formatted, with indentation 

 289     * levels indicated by colors in ascending sequence from the array, restarting at index 0 if the array is shorter than the maximum indentation level.

 290     * `colors` defaults to the static `COLORS` array. To disable, provide a value of `null`.

 291     * @return a string representation of an object literal, similar to the Node `util.inspect` call

 292     */

 293    public inspect(msg:any, depth=3, indent='    ', colors=COLORS):string {

 294        function _inspect(msg:any, depth:number, level:number, currIndent:string):string {

 295            if (msg === null)               { return 'null'; }

 296            if (msg === undefined)          { return 'undefined'; }

 297            if (typeof msg === 'function')  { return 'function'; }

 298            if (typeof msg === 'string')    { return `'${msg}'`; }

 299            if (typeof msg === 'object')    {

 300                if (depth<0) { return (msg.length===undefined)? '{...}' : '[...]'; }

 301                if (msg.map !== undefined) { 

 302                    return `[${msg.map((e:any)=>(e===undefined)?'':_inspect(e, depth-1, level+1, currIndent)).join(', ')}]`;

 303                 }

 304                const [prefix, postfix] = log.getPrePostfix(indent, level, currIndent);

 305                const cstart  = colors? `color:${colors[level % colors.length]};'>` : ''; 

 306                const cstop   = colors? `` : ''; 

 307                const lf      = colors? '
' : '\n';

 308                return `{${lf}` + Object.keys(msg).map(k => `${cstart}${prefix}${k}${postfix}${cstop}: ${

 309                        _inspect(msg[k], depth-1, level+1, currIndent+indent)

 310                    }`).join(`,${lf}`) + `${lf}${currIndent}}`;

 311            } 

 312            return msg.toString();

 313        }

 314        const log = this;

 315        if (colors) { indent = indent.replace(/ /g, ' '); }

 316        return _inspect(msg, depth===null? 999 : depth, 0, '');

 317    }

 318

 319    protected getPrePostfix(indent:string, level:number, currIndent:string):[string,string] {

 320        return [`${currIndent}${indent}`, ''];

 321    }
