   1/**

   2 * Logging convenience functions.

   3 * ## Usage

   4 * 

   5 * ### Using the global `Log` instance 

   6 * Settings in `Log.log` are shared across modules

   7 * ```

   8 * import { Log } from 'hsutil'; const log = Log.log;

   9 * log.level(Log.INFO);         // sets the reporting level on the global instance 

  10 * log.info('by the way:');     // -> 20160817 09:59:08.032 info by the way:

  11 * log.info(()=>'by the way:'); // -> 20160817 09:59:08.032 info by the way:

  12 * log.error('oh dear!');       // -> 20160817 09:59:08.045 error *** oh dear!

  13 * ```

  14 * 

  15 * ### Using a local `Log` instance 

  16 * Create a new `Log` instance to keep changes local to the current module:

  17 * ```

  18 * import { Log } from 'hsutil'; 

  19 * const log = new Log('myModule');

  20 * log.info('by the way:'); // -> 20160817 09:59:08.032 myModule info by the way:

  21 * log.error('oh dear!');   // -> 20160817 09:59:08.045 myModule error *** oh dear!

  22 * ```

  23 * 

  24 * ### Using the `format` template:

  25 * ```

  26 * log.format('%MMM %DD %hh%mm%ss');

  27 * log.info('by the way:');  // -> Aug 17 095908 info by the way:

  28 * log.error('oh dear!');    // -> Aug 17 095908 error *** oh dear!

  29 * ```

  30 * 

  31 * ### Change the module prefix:

  32 * ```

  33 * log.prefix('Main');

  34 * log.format('%hh%mm%ss');

  35 * log.info('by the way:');  // -> 09:59:08.032 Main info by the way:

  36 * log.error('oh dear!');    // -> 09:59:08.045 Main error *** oh dear!

  37 * ```

  38 * 

  39 * ### Setting the Log level locally for the module

  40 * ```

  41 * log.level(log.WARN);

  42 * log.info('this will not be reported');

  43 * ``` 

  44 * 

  45 * ### Setting the Log level globally

  46 * Will be used if no local level is set

  47 * ```

  48 * log.level(log.WARN, true);               // set global log level

  49 * log.info('this will be reported');       // because local level defaults to INFORM

  50 * log.level(null);                         // unset the local level

  51 * log.info('this will not be reported');   // because no local level is defined and global level is set to WARN

  52 * ``` 

  53 * 

  54 * ### log.inspect utility

  55 * returns a string representing a deep inspection of `myObj`.

  56 * ```

  57 * log.info(log.inspect(myObj));                    // prints the structure to first depth

  58 * log.info(log.inspect(myObj), {});                // prints the structure to inifinite depth

  59 * log.info(log.inspect(myObj), {colors:null});     // prints the structure without colors

  60 * log.info(myObj);                                 // equivalent to `log.info(log.inspect(myObj));`

  61 * ``` 

  62 * 

  63 * ## Reporting Levels:

  64 * -   {@link log.Log.DEBUG Log.DEBUG}

  65 * -   {@link log.Log.INFO  Log.INFO}

  66 * -   {@link log.Log.WARN  Log.WARN}

  67 * -   {@link log.Log.ERROR Log.ERROR}

  68 * 

  69 * ## Reporting methods

  70 * -   {@link log.Log.debug log.debug()}

  71 * -   {@link log.Log.info  log.info()}

  72 * -   {@link log.Log.warn  log.warn()}

  73 * -   {@link log.Log.error log.error()}

  74 * 

  75 * ## Configurations:

  76 * -   {@link log.Log.level   log.level()}

  77 * -   {@link log.Log.format  log.format()}

  78 * -   {@link log.Log.prefix  log.prefix()}

  79 * 

  80 * ## Inspection support:

  81 * -   {@link log.Log.inspect log.inspect()}

  82 */

 203    /**

 204     * sets the reporting level according to `newLevel`. 

 205     * Valid values are {@link log.DEBUG log.DEBUG}, {@link log.INFO log.INFO}, 

 206     * {@link log.WARN log.WARN}, {@link log.ERROR log.ERROR}, or `null`.

 207     * Subsequent reporting calls will be filtered such that only calls with an importance 

 208     * at least the same as `newLevel` will be written to the log.<br>

 209     * 

 210     * By default, `newLevel` sets the reporting level for the module only, and takes precedence over

 211     * any global level setting. Providing `null` as level value passes precedence back to the global setting.

 212     * This allows for simple per-module settings, for example in debugging scenarios.

 213     * @param newLevelSym the new reporting level to set. 

 214     * If omitted, the method returns the currently set reporting level. 

 215     * If set to `null`, the module's local reporting level defaults to the global reporting level; `setGlobalLevel` is ignored.

 216     * @param setGlobalLevel defaults to `false`. If `true`, sets the global reporting level for all modules. 

 217     * @return the new reporting level (DEBUG, INFO, WARN, ERROR)

 218     */

 219    public level(newLevelSym?:string, setGlobalLevel?:boolean):string {

 220        let newLevel = Log.levels[newLevelSym] || Log.globalLevel;  // new level is newLevelSym unless undefined

 221        let oldLevel = this.reportLevel || Log.globalLevel;         // old level is level unless undefined

 222        if (newLevelSym === undefined) {                            // do nothing, return current level

 223            newLevel = oldLevel;

 224        } else if (newLevelSym === null) {

 225            this.reportLevel = undefined;                           // deactivate local level

 226        } else if (Log.levels[newLevelSym]) { 

 227            if (setGlobalLevel) { Log.globalLevel = newLevel; }     // set new global level

 228                           else { this.reportLevel = newLevel; }    // set new local level

 229            const msg = `new ${setGlobalLevel? 'global' : this.reportPrefix} log level ${newLevel.desc.toUpperCase()} (was ${oldLevel.desc.toUpperCase()})`;

 230            (newLevel.sym === oldLevel.sym)? this.debug(msg) : this.info(msg);

 231        } else { 

 232            this.error(`unkown level ${newLevelSym}; log level remains ${oldLevel.sym}`);

 233        }

 234        return newLevel.sym;

 235    }

 296    /**

 297     * reports an error message to the log. 

 298     * The message will be reported to the log if `lvl` meets or exceeds the current reporting level.

 299     * @param lvl the reporting level of `msg`

 300     * @param msg the message to report. If `msg` is an object literal, a deep inspection will be printed. 

 301     * Else if `msg` is a function, it will be called to return the string to print or object to inspect. This 

 302     * avoids constructing a potentially expensive message string if the level is below the current reporting level.

 303     * Finally, if `msg` is a string, it will be printed with appropriate decoration, e.g. a date string and 

 304     * prefix.

 305     * @return the message printed

 306     */

 307    protected out(lvl:string, msg:any|any[], options:Msg): string {

 308        if (msg instanceof Array) return msg.map(m => this.out(lvl, m, options)).join('\n')

 309

 310        let lvlDesc:LevelDesc = Log.levels[lvl];

 311        const filterLevel = this.reportLevel || Log.globalLevel;

 312        if (lvlDesc.importance >= filterLevel.importance) {

 313            let line = '';

 314            switch(typeof msg) {

 315                case 'function': return this.out(lvl, msg(), options)

 316                case 'string':   line = msg; break;

 317                case 'object':   

 318                    if (msg.message) line += this.out(Log.ERROR, msg.message, options);

 319                    if (msg.stack) line += this.out(Log.ERROR, msg.stack, options);

 320                    else console.dir(msg);

 321                    return line

 322                default: line = this.inspect(msg); 

 323            }

 324            const dateStr = date(Log.dateFormat);

 325            if (msg.stack) { line = `${line}\n${msg.stack}`; }

 326            const headerParts = [dateStr, this.reportPrefix, lvlDesc.desc]

 327            this.output(options, headerParts, line);

 328            return `${headerParts.join(' ')} ${line}${(options.lf||'')}`;

 329        }

 330        return undefined;

 331    }

 333    /** 

 334     */

 335    protected output(options:Msg, headerParts:string[], line:string) {

 336        const color = options.color

 337        const lines = this.limitLength(line, options.maxLen)

 338        console.log(`%c${headerParts.join(' ')}%c ${lines.join('\n')}`, color.map(c =>COLOR[c]).join(' '), COLOR['clear']);

 339    }

 341    protected limitLength(line:string, maxLen:number) {

 342        const lines = line.split('\n')

 343        maxLen = maxLen || Log.maxLength || Log.transientLength;

 344        if (maxLen>0) {

 345            lines.forEach((l,i) => lines[i] = l.length <= maxLen? 

 346                lines[i].padEnd(maxLen, ' ') :

 347                `${line.slice(0, maxLen/2-2)}...${line.slice(-maxLen/2+2)}`)

 348        }

 349        return lines

 350    }

 385    public config(cfg:{colors?:boolean, format?:string, level?:string }) {

 386        if (cfg.format!==undefined) { Log.log.format(cfg.format); }     // e.g. '%YYYY%MM%DD %hh:%mm:%ss.%jjj'

 387        if (cfg.level!==undefined)  { this.level(cfg.level); }          // e.g. INFO

 388        if (cfg.colors!==undefined) { Log.colors = cfg.colors; }    // true / false

 389    }

 391    /**

 392     * Usage: 

 393     * - `log.info(log.inspect(struct))`     -> `depth=0`, `indent='   '`, with html-coded colors

 394     * - `log.info(log.inspect(struct, {}))` -> `depth=-1` (inf), `indent='   '`, with html-coded colors

 395     * - `log.info(log.inspect(struct, {colors:null}))` -> no colors

 396     * <p>

 397     * The call returns a raw formatted text string, or a HTML formatted string if `colors` is defined.

 398     * @param struct the object literal to inspect

 399     * @param depth depth of recursion, defaults to 3. Use `null` or negative values for infinite depth

 400     * @param indent the indent string to use, will accumulate for each level of indentation, defaults to four spaces.

 401     * @param colors an array of `css` color values: if defined, the output will be `HTML` formatted, with indentation 

 402     * levels indicated by colors in ascending sequence from the array, restarting at index 0 if the array is shorter than the maximum indentation level.

 403     * `colors` defaults to the static `COLORS` array. To disable, provide a value of `null`.

 404     * @return a string representation of an object literal, similar to the Node `util.inspect` call

 405     */

 406    public inspect(msg:any, {depth=-1, indent='   ', colors=Log.INDENT_COLORS}={depth:0, indent:'   ', colors:Log.INDENT_COLORS}):string {

 407        function _inspect(msg:any, depth:number, level:number, currIndent:string):string {

 408            if (msg === null)               { return 'null'; }

 409            if (msg === undefined)          { return 'undefined'; }

 410            if (typeof msg === 'function')  { return 'function'; }

 411            if (typeof msg === 'string')    { return `'${msg}'`; }

 412            if (typeof msg === 'object')    {

 413                if (depth<0) { 

 414                    return (msg.length===undefined)? '{...}' : '[...]'; 

 415                }

 416                if (msg.map !== undefined) { // --> an array

 417                    return `[${msg.map((e:any)=>(e===undefined)?'':_inspect(e, depth-1, level+1, currIndent)).join(', ')}]`;

 418                }

 419                const [prefix, postfix, lf, postIndent] = aLog.getPrePostfix(indent, level, currIndent, colors);

 420                return `{${lf}` + Object.keys(msg).map(k => `${prefix}${k}${postfix}: ${

 421                    _inspect(msg[k], depth-1, level+1, currIndent+indent)

 422                }`).join(`,${lf}`) + `${lf}${postIndent}}`;

 423            } 

 424            return msg.toString();

 425        }

 426        const aLog = this;

 427        // if (colors) { indent = indent.replace(/ /g, ' '); }

 428        return _inspect(msg, (depth!==undefined && depth!==null && depth>=0) ? depth : 999, 0, '');

 429    }

 430

 431    protected getPrePostfix(indent:string, level:number, currIndent:string, colors:string[]):[string,string,string,string] {

 432        let cstart = '', cstop = '', lf = '\n';

 433        if (colors) {

 434            indent = indent.replace(/ /g, ' ');

 435            currIndent = currIndent.replace(/ /g, ' ');

 436            // try <color name> first, then assume '#...' code

 437            const color = COLOR[colors[level % colors.length]] || colors[level % colors.length];

 438            cstart  = `<b><span style='${color}'>`; 

 439            cstop   = `</span></b>`; 

 440            lf      = '<br>';

 441        }

 442        return [`${currIndent}${indent}${cstart}`, `${cstop}`, colors? '<br>' : '\n', currIndent];

 443    }
