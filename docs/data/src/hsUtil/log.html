   1/**

   2 * Logging convenience functions.

   3 * ## Usage

   4 * 

   5 * ### Using the global log object 

   6 * Settings in `log` are shared across modules

   7 * ```

   8 * import { log } from 'hsutil'; 

   9 * log.info('by the way:'); // -> 20160817 09:59:08.032 info by the way:

  10 * log.error('oh dear!');   // -> 20160817 09:59:08.045 error *** oh dear!

  11 * ```

  12 * 

  13 * ### Using a local log object 

  14 * Settings in `log` remain local to the module 

  15 * ```

  16 * import { log as gLog } from 'hsutil'; const log = gLog('myModule')

  17 * log.info('by the way:'); // -> 20160817 09:59:08.032 myModule info by the way:

  18 * log.error('oh dear!');   // -> 20160817 09:59:08.045 myModule error *** oh dear!

  19 * ```

  20 * 

  21 * ### Using the `format` template:

  22 * ```

  23 * log.format('%MMM %DD %hh%mm%ss');

  24 * log.info('by the way:');  // -> Aug 17 095908 info by the way:

  25 * log.error('oh dear!');    // -> Aug 17 095908 error *** oh dear!

  26 * ```

  27 * 

  28 * ### Change the module prefix:

  29 * ```

  30 * log.prefix('Main');

  31 * log.format('%hh%mm%ss');

  32 * log.info('by the way:');  // -> 09:59:08.032 Main info by the way:

  33 * log.error('oh dear!');    // -> 09:59:08.045 Main error *** oh dear!

  34 * ```

  35 * 

  36 * ### Setting the Log level locally for the module

  37 * ```

  38 * log.level(log.WARN);

  39 * log.info('this will not be reported');

  40 * ``` 

  41 * 

  42 * ### Setting the Log level globally

  43 * Will be used if no local level is set

  44 * ```

  45 * log.level(log.WARN, true);               // set global log level

  46 * log.info('this will be reported');       // because local level defaults to INFORM

  47 * log.level(null);                         // unset the local level

  48 * log.info('this will not be reported');   // because no local level is defined and global level is set to WARN

  49 * ``` 

  50 * 

  51 * ### log.inspect utility

  52 * returns a string representing a deep inspection of `myObj`.

  53 * ```

  54 * log.info(log.inspect(myObj, null));      // prints the structure to inifinite depth

  55 * ``` 

  56 * 

  57 * ## Reporting Levels:

  58 * -   {@link log.DEBUG log.DEBUG}

  59 * -   {@link log.INFO  log.INFO}

  60 * -   {@link log.WARN  log.WARN}

  61 * -   {@link log.ERROR log.ERROR}

  62 * 

  63 * ## Reporting methods

  64 * -   {@link log.LogType.debug log.debug()}

  65 * -   {@link log.LogType.info  log.info()}

  66 * -   {@link log.LogType.warn  log.warn()}

  67 * -   {@link log.LogType.error log.error()}

  68 * 

  69 * ## Configurations:

  70 * -   {@link log.LogType.level   log.level()}

  71 * -   {@link log.LogType.format  log.format()}

  72 * -   {@link log.LogType.prefix  log.prefix()}

  73 * -   {@link log.LogType.logFile log.logFile()}

  74 * 

  75 * ## Inspection support:

  76 * -   {@link log.LogType.inspect log.inspect()}

  77 */

 139    /**

 140     * sets the reporting level according to `newLevel`. 

 141     * Valid values are {@link log.DEBUG log.DEBUG}, {@link log.INFO log.INFO}, 

 142     * {@link log.WARN log.WARN}, {@link log.ERROR log.ERROR}, or `null`.

 143     * Subsequent reporting calls will be filtered such that only calls with an importance 

 144     * at least the same as `newLevel` will be written to the log.

 145     * 

 146     * By default, `newLevel` sets the reporting level for the module only, and takes precedence over

 147     * any global level setting. Providing `null` as level value passes precedence back to the global setting.

 148     * This allows for simple per-module settings, for example in debugging scenarios.

 149     * @param newLevelSym the new reporting level to set. 

 150     * If omitted, the method returns the currently set reporting level. 

 151     * If set to `null`, the module's local reporting level defaults to the global reporting level; `setGlobalLevel` is ignored.

 152     * @param setGlobalLevel defaults to `false`. If `true`, sets the global reporting level for all modules. 

 153     * @return the new reporting level (DEBUG, INFO, WARN, ERROR)

 154     */

 155    level(newLevelSym?:symbol, setGlobalLevel?:boolean):symbol;

 284    function level(newLevelSym?:symbol, setGlobalLevel=false):symbol {

 285        let newLevel = gLevels[newLevelSym] || gGlobalLevel;    // new level is newLevelSym unless undefined

 286        let oldLevel = context.level || gGlobalLevel;           // old level is level unless undefined

 287        if (newLevelSym === undefined) {                        // do nothing, return current level

 288            newLevel = oldLevel;

 289        } else if (newLevelSym === null) {

 290            context.level = undefined;                          // deactivate local level

 291        } else if (gLevels[newLevelSym]) { 

 292            if (setGlobalLevel) { gGlobalLevel = newLevel; }    // set new global level

 293                           else { context.level = newLevel; }     // set new local level

 294            const msg = `new ${setGlobalLevel? 'global' : context.prefix} log level ${newLevel.desc.toUpperCase()} (was ${oldLevel.desc.toUpperCase()})`;

 295            out((newLevel.sym === oldLevel.sym)?DEBUG : INFO, msg);

 296        } else { 

 297            out(ERROR, `unkown level ${newLevelSym.toString()}; log level remains ${oldLevel.sym.toString()}`);

 298        }

 299        return newLevel.sym;

 300    }

 318    async function out(lvl:symbol, msg:any, log2File=true): Promise {

 319        const colors = { [ERROR]: color.red+color.bold, [WARN]: color.yellow+color.bold, [DEBUG]: color.blue, [INFO]: color.green };

 320        let desc = gLevels[lvl];

 321        const filterLevel = context.level || gGlobalLevel;

 322        if (desc.importance >= filterLevel.importance) {

 323            const dateStr = date(gDateFormat);

 324            let line = (typeof msg === 'string')? msg : inspect(msg, 0);

 325            const logLine   =                    `${dateStr} ${context.prefix} ${desc.desc} ${line}`;

 326            const colorLine = `${colors[lvl]||''} ${dateStr} ${context.prefix} ${desc.desc} ${color.clear} ${line}`;

 327            console.log(gColors? colorLine : logLine);

 328            if (msg && msg.stack) { console.log(msg.stack); }

 329            if (gLogFile && log2File) {

 330                return await context.logToFile(date(gLogFile), logLine);

 331            }

 332        }

 333        return undefined;

 334    }

 336    async function logFile(file?:string):Promise {

 337        if (file === null) {                    // disable logging in file

 338            gLogFile = undefined; 

 339            return await info("disabling logfile");

 340        } else if (file === undefined) {        // leave gLogFile unchanged, return promise for logfile name

 341            return date(gLogFile);

 342        } else if (file.indexOf('/')>=0) { 

 343            return await context.pathExists(file)

 344                .then(async (exists:boolean) => { 

 345                    if (!exists) {

 346                        gLogFile = undefined;

 347                        return await warn(`path '${file}' doesn't exists; logfile disabled`);

 348                    }

 349                    gLogFile = file;

 350                    return await info("now logging to file " + date(file));

 351                })

 352                .catch(async () => { 

 353                    gLogFile = undefined; 

 354                    return await error(`checking path ${file}; logfile disabled`);

 355                });

 356        } else if (file === '') {

 357            file = 'log-%YYYY-%MM-%DD.txt';

 358        } else {

 359        }

 360        gLogFile=file;

 361        return await info(gLogFile? `now logging to file ${date(gLogFile)}` : 'logfile disbaled');

 362    }

 370    function inspect(msg:any, depth=3, indent='   ', colors?:string[]):string {

 371        function _inspect(msg:any, depth:number, level:number, currIndent:string):string {

 372            if (msg === null)               { return 'null'; }

 373            if (msg === undefined)          { return 'undefined'; }

 374            if (typeof msg === 'function')  { return 'function'; }

 375            if (typeof msg === 'string')    { return `'${msg}'`; }

 376            if (typeof msg === 'object')    {

 377                if (depth<0) { return (msg.length===undefined)? '{...}' : '[...]'; }

 378                if (msg.length !== undefined) { 

 379                    return `[${msg.map((e:any)=>(e===undefined)?'':_inspect(e, depth-1, level+1, currIndent)).join(', ')}]`;

 380                 }

 381                const c  = colors? `color:${colors[level % colors.length]};'>` : '';

 382                const prefix = `${c}${currIndent}${indent}`;

 383                const postfix = colors? '' : '';

 384                return '{\n' + Object.keys(msg).map(k => `${prefix}${k}${postfix}: ${

 385                        _inspect(msg[k], depth-1, level+1, currIndent+indent)

 386                    }`).join(',\n') + `\n${currIndent}}`;

 387            } 

 388            return msg.toString();

 389        }

 390        if (colors) { indent = indent.replace(/ /g, ' '); }

 391        return _inspect(msg, depth===null? 999 : depth, 0, '');

 392    }

 393

 394    const newLog:any = (prefix:string, logToFile:ltfType=context.logToFile, pathExists:peType=context.pathExists) => create(prefix, logToFile, pathExists);

 395  

 396    newLog.DEBUG    = DEBUG;

 397    newLog.INFO     = INFO;

 398    newLog.WARN     = WARN;

 399    newLog.ERROR    = ERROR;

 400    newLog.level    = level;

 401    newLog.debug    = debug;

 402    newLog.info     = info;

 403    newLog.warn     = warn;

 404    newLog.error    = error;

 405    newLog.format   = format;

 406    newLog.prefix   = prefix;

 407    newLog.out      = out;

 408    newLog.logFile  = logFile;

 409    newLog.config   = config;

 410    newLog.inspect  = inspect;

 411    return newLog;
