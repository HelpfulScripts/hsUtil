   7/**
   8 * @description timeout promise for use in `Promise.race()`.
   9 * @param {number} ms the milliseconds to wait before rejecting
  10 * @return {Promise} a Promise that rejects after `ms` 
  11 */
  12export function timeout(ms:number):Promise { 
  13    return new Promise((resolve, reject) => { setTimeout(reject, ms); }); 
  14}

  16/**
  17 * @description delays a promise, passing the parameter 
  18 * received from the calling promise down to the resolving promise.
  19 * 
  20 * **Usage 1:** a delay within the `then` chain:
  21 * ```
  22 * 
  23 *    .then(...)
  24 *    .then(delay(10))
  25 *    .then(...)
  26 *    .catch(...)
  27 * ```
  28 * **Usage 2:** an initial delay
  29 * ```
  30 * delay(10)()
  31 *    .then(...)
  32 *    .catch(...)
  33 * ```
  34 * @param number ms the milliseconds to wait before resolving
  35 * @return a `Promise` that resolves after `ms` 
  36 */
  37export function delay(ms:number)   { 
  38    return (args?:T):Promise => {
  39        return new Promise((resolve:(args:T)=>void) => { 
  40            setTimeout(() => { resolve(args); }, ms); 
  41        }); 
  42    };
  43}

  45/**
  46 * @description ensures that a functions in a sequence are not executed faster than a preset minimum delay.
  47 * 
  48 * **Usage:** 
  49 * ```
  50 * const q = new PacingQueue(100); // 100ms between calls
  51 * ...
  52 * q.add((ms) => `I have been called after ${ms}ms`;})
  53 *    .then((result) => console.log(result));     // prints: I have been called after ***ms`
  54 * ```
  55 */
  56export class Pace {
  57    private pace:number;    // the pace of calls in ms
  58    private waitUntil = 0;  // the earliest time for the next call
  59
  60    /**
  61     * @param delay the minimum number of milliseconds between executions of 
  62     * two registered functions; defaults to 100;
  63     */
  64    constructor(pace=100) {
  65        this.pace = pace+5; // add 5ms margin. delay() may trigger a millisecond or two early
  66    }
  67
  68    /**
  69     * adds the function to the queue. After an appropriate time has passed, 
  70     * the function will be called with the delay in ms as parameter.
  71     * @param fn 
  72     * @return a promise that resolved to the result of the function
  73     */
  74    add(fn: (msSinceAdding:number) => any):Promise {
  75        const addTime = Date.now();
  76        if (this.waitUntil < addTime) {
  77            this.waitUntil = addTime + this.pace;
  78            return Promise.resolve(fn(Date.now()-addTime));
  79        } else {
  80            const diff = this.waitUntil - addTime;
  81            this.waitUntil += this.pace + 5;
  82            return Promise.resolve().then(delay(diff)).then(() => {
  83                return fn(Date.now()-addTime);
  84            });
  85        }
  86    }
  87}

  90/**
  91 * Sequentially calls the provided `tasks` in a `.then()` chain of promises, 
  92 * guaranteeing the order of execution per index order in the array. 
  93 * Each task can return a result, or a promise for a result.
  94 * @param tasks an array of task calls to execute in sequence. Each call will pass the array of results for tasks so far executed.
  95 * @param initialResult optional initial array to collect the task results.
  96 */
  97export function promiseChain(tasks:((results:T[])=>T|Promise)[], initialResult:T[]=[]): Promise {
  98    return tasks.reduce((chain:Promise, task:(result:T[])=>T|Promise): Promise =>
  99        // execute a task by chaining it to the previous ones via `.then()`
 100        chain.then((_results:T[]) => Promise.resolve(task(_results)).then((r:T) => {
 101            // add the task's result to the `results` array
 102            _results.push(r);
 103            return _results;
 104        })), 
 105        Promise.resolve(initialResult)
 106    );
 107}
