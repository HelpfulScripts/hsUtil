  20 * @return a `Promise` that resolves after `ms`

  22export function delay(ms:number)   { 
  23    return (args:T):Promise => {
  24        return new Promise((resolve:(args:T)=>void) => { 
  25            setTimeout(() => { resolve(args); }, ms); 
  26        }); 
  27    };
